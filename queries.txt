CREATE TABLE users (
	user varchar,
	e_mail varchar,
	biography varchar,
	avatar_name varchar,
	password varchar
);

CREATE TABLE user_log (
	user varchar,
	user_action varchar,
	time_stamp varchar
);

CREATE TABLE posts (
	id integer,
	user varchar,
	description varchar,
	votes integer,
	time_stamp varchar
);

CREATE TABLE comments (
	posts_id integer,
	id integer,
	user varchar,
	description varchar
);

CREATE TABLE user_vote ( 
	user VARCHAR, posts_id INT , time_stamp VARCHAR
	);
